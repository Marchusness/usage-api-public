# Auth

```ts
import { auth } from "usage-api";

const res = auth(
  process.env.USAGE_API_KEY,
  usageStoreApiKey, // api key from the user
  {
    blockBehavior: "ignore" | "block", // default block
  }
);

type res =
  | {
      success: true;
    }
  | {
      success: false;
      message: string;
    };
```
