# Create API Key

```ts
import { deleteApiKey } from "usage-api";

const res = deleteApiKey(process.env.USAGE_API_KEY, {
    usageStoreId: "my-usage-store-id", // required
    apiKeyId: "my-api-key-id" // required
});

type res = {
    success: true,
} | {
    success: false,
    message: string
}
```
