# Create API Key

```ts
import { deleteApiKey } from "usage-api";

const res = deleteApiKey(process.env.USAGE_API_KEY, {
  usageStoreId: string, // required
  apiKeyId: string, // required
});

type res =
  | {
      success: true;
    }
  | {
      success: false;
      message: string;
    };
```
